### 2.3 集成到后端代码

**创建 `src/config/sentry.ts`：**

```typescript
import * as Sentry from '@sentry/node';
import { ProfilingIntegration } from '@sentry/profiling-node';

export function initSentry() {
  if (!process.env.SENTRY_DSN) {
    console.log('Sentry DSN not configured, skipping initialization');
    return;
  }

  Sentry.init({
    dsn: process.env.SENTRY_DSN,
    environment: process.env.SENTRY_ENVIRONMENT || 'development',

    // 性能监控
    tracesSampleRate: 0.1, // 10% 的请求被追踪

    // 性能分析
    profilesSampleRate: 0.1,
    integrations: [
      new ProfilingIntegration(),
    ],

    // 忽略某些错误
    ignoreErrors: [
      'Non-Error promise rejection captured',
      'Network request failed',
    ],

    // 添加上下文信息
    beforeSend(event, hint) {
      // 添加自定义数据
      if (event.request) {
        event.request.headers = {
          ...event.request.headers,
          // 移除敏感信息
          authorization: '[Filtered]',
        };
      }
      return event;
    },
  });

  console.log('✅ Sentry initialized');
}
```

### 2.4 在 server.ts 中初始化

**编辑 `src/server.ts`，在文件最顶部添加：**

```typescript
// 必须在最顶部，在其他 import 之前！
import { initSentry } from './config/sentry';
import * as Sentry from '@sentry/node';

initSentry();

// ... 其他 imports
import express from 'express';
// ...

const app = express();

// 在所有中间件之前
app.use(Sentry.Handlers.requestHandler());
app.use(Sentry.Handlers.tracingHandler());

// ... 你的路由和中间件 ...

// 在所有路由之后，但在错误处理之前
app.use(Sentry.Handlers.errorHandler());

// 你的错误处理中间件
app.use(errorHandler);
```

### 2.5 手动捕获错误

### 3.3 初始化 Sentry

**编辑 `src/main.tsx`（或 `src/index.tsx`）：**

```typescript
import React from 'react';
import ReactDOM from 'react-dom/client';
import * as Sentry from '@sentry/react';
import App from './App';

// 初始化 Sentry
if (import.meta.env.VITE_SENTRY_DSN) {
  Sentry.init({
    dsn: import.meta.env.VITE_SENTRY_DSN,
    environment: import.meta.env.VITE_SENTRY_ENVIRONMENT || 'development',

    // React Router 集成
    integrations: [
      Sentry.browserTracingIntegration(),
      Sentry.replayIntegration({
        maskAllText: false,
        blockAllMedia: false,
      }),
    ],

    // 性能监控
    tracesSampleRate: 0.1,

    // Session Replay
    replaysSessionSampleRate: 0.1,
    replaysOnErrorSampleRate: 1.0,

    // 忽略某些错误
    ignoreErrors: [
      'ResizeObserver loop limit exceeded',
      'Non-Error promise rejection captured',
    ],

    // 添加用户信息
    beforeSend(event, hint) {
      // 可以添加用户信息
      const user = localStorage.getItem('user');
      if (user) {
        event.user = JSON.parse(user);
      }
      return event;
    },
  });

  console.log('✅ Sentry initialized');
}

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <App />
  </React.StrictMode>
);
```

### 3.4 添加 Error Boundary

**创建 `src/components/ErrorBoundary.tsx`：**

```typescript
import * as Sentry from '@sentry/react';
import { Component, ReactNode } from 'react';

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
}

class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError() {
    return { hasError: true };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    Sentry.captureException(error, { extra: errorInfo });
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="min-h-screen flex items-center justify-center bg-gray-50">
          <div className="text-center">
            <h1 className="text-4xl font-bold text-gray-900 mb-4">
              出错了 😔
            </h1>
            <p className="text-gray-600 mb-8">
              我们已经收到错误报告，会尽快修复！
            </p>
            <button
              onClick={() => window.location.href = '/'}
              className="px-6 py-3 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
            >
              返回首页
            </button>
          </div>
        </div>
      );
    }

    return this.props.children;
  }
}

export default Sentry.withErrorBoundary(ErrorBoundary, {
  fallback: <div>出错了...</div>,
});
```

**在 App.tsx 中使用：**

```typescript
import ErrorBoundary from './components/ErrorBoundary';

function App() {
  return (
    <ErrorBoundary>
      {/* 你的应用 */}
    </ErrorBoundary>
  );
}
```

## 📊 Sentry 最佳实践

### 1. 添加用户信息

```typescript
// 用户登录后
Sentry.setUser({
  id: user.id,
  email: user.email,
  username: user.username,
});

// 用户登出后
Sentry.setUser(null);
```

### 2. 添加面包屑（Breadcrumbs）

```typescript
Sentry.addBreadcrumb({
  category: 'workflow',
  message: 'User started workflow execution',
  level: 'info',
  data: {
    workflowId: '123',
  },
});
```

### 3. 设置上下文

```typescript
// 全局上下文
Sentry.setContext('app', {
  version: '1.0.0',
  build: '2025-11-03',
});

// 请求上下文
Sentry.setContext('request', {
  userId: req.user?.id,
  ip: req.ip,
});
```

### 4. 性能监控

```typescript
// 手动创建事务
const transaction = Sentry.startTransaction({
  name: 'Workflow Execution',
  op: 'workflow.execute',
});

try {
  // 你的代码
  const span = transaction.startChild({
    op: 'ai.generate',
    description: 'Generate workflow steps',
  });

  await generateWorkflow();

  span.finish();
} finally {
  transaction.finish();
}
```

## 🔍 监控指标

### 查看的关键指标

1. **错误率**：每小时新错误数量
2. **用户影响**：多少用户遇到错误
3. **响应时间**：API 响应速度
4. **崩溃率**：应用崩溃频率

## 🚨 告警规则建议

### 关键告警

1. **高频错误**
   - 条件：同一错误 1小时内出现 > 10次
   - 动作：立即邮件 + Slack

2. **新错误**
   - 条件：首次出现的错误
   - 动作：邮件通知

3. **性能下降**
   - 条件：响应时间 > 5秒
   - 动作：邮件通知

4. **用户影响**
   - 条件：错误影响 > 100 用户
   - 动作：紧急通知
